$SQL = @"
UPDATE public.modules
SET "s3Key" = 'videos/' || id || '.mp4'
WHERE "s3Key" IS NULL OR "s3Key" = '';

UPDATE public.modules

WHERE "stepsKey" IS NULL OR "stepsKey" = '';
